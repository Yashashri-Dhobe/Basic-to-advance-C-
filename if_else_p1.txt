#include<iostream>
using namespace std;

int main ()

{
    int num; 
    cout << "enter a number\n";
    cin >> num;
    
    if (num > 0){
        cout << "entered a number is positive\n " <<num ;
        
    } else if (num < 0) {
        cout << "entered a number is negative\n " << num ;
    } else {
        cout << "entered a number is zero " <<num ;
    }
    
    return 0;
}
